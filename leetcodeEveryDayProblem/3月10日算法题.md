class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<string> chess(n, string(n, '.')); //创建一个全是空格的棋盘数组
        result.clear();
        dfs(n, 0, chess);
        return result;
    }
private:
    vector<vector<string>> result;
    void dfs(int n, int row, vector<string>& chess){
        //如果遍历到棋盘最后一行，就把当前有效棋盘存入结果
        if(n == row){
            result.push_back(chess);
            return;
        }
		//遍历棋盘当前这一行，判断每个棋盘位置是否能够放皇后
        for(int i = 0; i < n; i++){
            //如果可以放皇后，就在当前位置变为Q，继续递归遍历下一行
            if(isValid(row, i, n, chess)){
                chess[row][i] = 'Q';
                dfs(n, row+1, chess);
                chess[row][i] = '.';
            }
        }
        //如果遍历完一整行也没有找到有效皇后位置，说明上一行皇后位置不对，继续在上一行后续位置寻找有效皇后			位置
    }
	//判断当前位置是否是皇后的有效位置
    bool isValid(int row, int col, int n, const vector<string>& chess){
        //判断与当前位置同列的位置上是否有皇后
        for(int i = 0; i < row; i++){
            if(chess[i][col] == 'Q'){
                return false;
            }
        }
		//判断当前位置左上45°角对角线上是否有皇后
        for(int i = row-1, j = col-1; i >= 0 && j >= 0; i--,j--){
            if(chess[i][j] == 'Q'){
                return false;
            }
        }
		//判断当前位置右上45°角对角线上是否有皇后
        for(int i = row-1, j = col+1; i >= 0 && j < n; i--, j++){
            if(chess[i][j] == 'Q'){
                return false;
            }
        }
        //如果上面都没有皇后，说明该位置是皇后有效位置
        return true;
    }

};

## [37. 解数独 - 力扣（LeetCode）](https://leetcode.cn/problems/sudoku-solver/description/)

class Solution {
public:
    void solveSudoku(vector<vector<char>>& board) {
        dfs(board, 0, 0);
    }
private:
    bool dfs(vector<vector<char>>& board, int row, int col){
        //如果当前行遍历完，就遍历下一行
        if (col == 9) return dfs(board, row + 1, 0);
        //如果最后一行遍历完，说明已解完数独，返回true
        if (row == 9) return true;
        //如果当前位置已有数字，就继续解当前行的下一个位置
        if (board[row][col] != '.') return dfs(board, row, col + 1);
		
        //遍历1到9，寻找有效的数字作为当前位置的解
        for (char c = '1'; c <= '9'; ++c) {
            if (isValid(row, col, c, board)) {
                board[row][col] = c; //如果该数字符合当前位置的解，就将该数字作为当前位置的解
                if (dfs(board, row, col + 1)) return true; //已找到整个数独的唯一正确解
                board[row][col] = '.';//如果后续位置没有正确的解，继续在这个解之后寻找当前位置的解
            }
        }
        
        //1到9都没有符合当前位置的解，说明上一个位置的解有问题，返回false，并在上一位置继续寻找符合条件			的解
        return false;
    }
	//判断该数字是否是当前位置的正确解
    bool isValid(int row, int col, char c, const vector<vector<char>>& board){
        //判断当前行是否有数字和该数字相同
        for(int i = 0; i < 9; i++){
            if(board[row][i] == c){
                return false;
            }
        }
		//判断当前行是否有数字和该数字相同
        for(int i = 0; i < 9; i++){
            if(board[i][col] == c){
                return false;
            }
        }
		//计算当前位置所在3*3宫格左上角的行坐标
        int starti = row/3*3;
        //计算当前位置所在3*3宫格左上角的列坐标
        int startj = col/3*3;
		//判断当前位置所在3*3宫格是否有数字和该数字相同
        for(int i = starti; i < starti+3; i++){
            for(int j = startj; j < startj+3; j++){
                //如果有相同的，说明该数字不是当前位置的正确解
                if(board[i][j] == c){
                    return false;
                }
            }
        }
		//上述都没有找到和该数字相同的数字，说明该数字是当前位置的正确解
        return true;
    }
};
